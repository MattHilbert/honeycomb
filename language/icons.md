<h2 id="Fundamentals">Fundamentals</h2>



    <h3>Choose commonly recognised icons whenever possible</h3>

<br>
<br>
<br>

    <img src="{{ site.baseurl }}/assets/images/icon-padding.png" alt="Icon padding" title="Icon padding" class="js-svg">
